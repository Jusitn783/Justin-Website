Code Example from the Game

Here's a snippet from our game that adds a ticking clock sound effect in Unity:


using UnityEngine;

public class ClockTicking : MonoBehaviour
{
    public AudioSource audioSource;
    public AudioClip tickingSound;
    public float interval = 1f;

    void Start()
    {
        if (audioSource == null)
        {
            audioSource = GetComponent();
        }

        if (audioSource != null && tickingSound != null)
        {
            audioSource.spatialBlend = 1f; 
            audioSource.rolloffMode = AudioRolloffMode.Logarithmic; 
            audioSource.maxDistance = 15f; 

            InvokeRepeating("PlayTickingSound", 0f, interval);
        }
        else
        {
            Debug.LogWarning("AudioSource or ticking sound missing on " + gameObject.name);
        }
    }

    void PlayTickingSound()
    {
        audioSource.PlayOneShot(tickingSound);
    }
}
